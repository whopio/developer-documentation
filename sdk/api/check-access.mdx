---
title: Check Access
description: Check if a user has access to an experience, access pass, or company.
---

<CodeGroup>

```typescript @whop/api
import { checkAccess } from "@whop/api";

// Check access to experience
const experienceAccess: AccessResult = await checkAccess.experience({ 
  experienceId: "experience_id" 
});

// Check access to access pass
const accessPassAccess: AccessResult = await checkAccess.accessPass({ 
  accessPassId: "access_pass_id" 
});

// Check access to company
const companyAccess: AccessResult = await checkAccess.company({ 
  companyId: "company_id" 
});
```

```typescript GraphQL
// Check experience access
const experienceQuery = `
  query CheckExperienceAccess($experienceId: ID!) {
    hasAccessToExperience(experienceId: $experienceId) {
      hasAccess
      accessLevel
    }
  }
`;

// Check access pass access
const accessPassQuery = `
  query CheckAccessPassAccess($accessPassId: ID!) {
    hasAccessToAccessPass(accessPassId: $accessPassId) {
      hasAccess
      accessLevel
    }
  }
`;

// Check company access
const companyQuery = `
  query CheckCompanyAccess($companyId: ID!) {
    hasAccessToCompany(companyId: $companyId) {
      hasAccess
      accessLevel
    }
  }
`;

const response = await fetch("https://api.whop.com/api/v2", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "Authorization": `Bearer ${process.env.WHOP_API_KEY}`
  },
  body: JSON.stringify({
    query: experienceQuery, // or accessPassQuery or companyQuery
    variables: {
      experienceId: "experience_id" // or accessPassId or companyId
    }
  })
});

const { data } = await response.json();
const accessResult = data.hasAccessToExperience; // or hasAccessToAccessPass or hasAccessToCompany
```
