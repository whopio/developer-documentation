---
title: Get Access Pass
description: Get the access pass information for the current user.
---

<CodeGroup>

```typescript @whop/api
import { getAccessPass } from "@whop/api";

const response = await fetch("https://api.whop.com/api/v2", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "Authorization": `Bearer ${process.env.WHOP_API_KEY}`
  },
  body: JSON.stringify({
    query,
    variables: {}
  })
});

const { data } = await response.json();
const accessPass: AccessPass = data.hasAccessToExperience;
```
